```python
def _on_claude_apply(self, param_dict: dict):
    """Apply Claude-suggested parameters via CrossfadeEngine with undo support."""
    # Create a compound undo command for all params
    self._undo_stack.beginMacro("Apply Claude Suggestion")

    for param_name, value in param_dict.items():
        old_value = self._prev_values.get(param_name, value)
        # Push undo command
        cmd = ParameterChangeCommand(
            param_name=param_name,
            old_value=old_value,
            new_value=value,
            apply_fn=self._apply_param,  # or whatever the existing param apply function is
            merge_id=-1,  # No merging for Claude apply
        )
        self._undo_stack.push(cmd)

    self._undo_stack.endMacro()
```
