**Step 3: Add config file management to settings.py**
Add functions at the end of settings.py:
```python
import json
from pathlib import Path

_CONFIG_DIR = Path.home() / ".apollo7"
_CONFIG_FILE = _CONFIG_DIR / "config.json"

def load_api_key() -> str | None:
    """Load Anthropic API key from config file or environment."""
    # Environment variable takes priority
    env_key = os.environ.get("APOLLO7_CLAUDE_API_KEY")
    if env_key:
        return env_key
    # Try config file
    if _CONFIG_FILE.exists():
        try:
            data = json.loads(_CONFIG_FILE.read_text())
            return data.get("anthropic_api_key")
        except (json.JSONDecodeError, OSError):
            pass
    return None

def save_api_key(key: str) -> None:
    """Save Anthropic API key to config file."""
    _CONFIG_DIR.mkdir(parents=True, exist_ok=True)
    config = {}
    if _CONFIG_FILE.exists():
        try:
            config = json.loads(_CONFIG_FILE.read_text())
        except (json.JSONDecodeError, OSError):
            pass
    config["anthropic_api_key"] = key
    _CONFIG_FILE.write_text(json.dumps(config, indent=2))
    # Restrictive permissions (best effort on Windows)
    try:
        _CONFIG_FILE.chmod(0o600)
    except OSError:
        pass
```
